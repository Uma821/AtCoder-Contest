```C++
#include <iostream>
#include <vector>
#include <string>
using namespace std;
bool func(string S) {
  int l=0, r=S.size()-1;
  while(l<r) {
    if(S[l]!=S[r])
      return false;
    ++l;--r;
  }
  return true;
}
int main() {
  int N;
  cin >> N;
  vector<string> S(N);
  for(auto &&s: S)
    cin >> s;
  
  for(int i=0; i<N-1; ++i) {
    for(int j=i+1; j<N; ++j) {
      if(func(S[i]+S[j]) || func(S[j]+S[i])) {
        cout << "Yes\n";
        return 0;
      }
    }
  }
  cout << "No\n";
}
```
