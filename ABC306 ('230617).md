### 自分の回答
```C++
#include <iostream>
using namespace std;
int main() {
  unsigned long long Ans = 0;
  int A[64];
  for(auto&& a: A)
    cin >> a;
  for(int i=63; i>=0; --i) {
    Ans <<= 1;
    Ans |= A[i];
  }
  cout << Ans << endl;
}
```
